function initSubjectScopeFilters(form) { const root = form.querySelector("[data-subject-scope-root]"); if (!root) return; const modeSelect = root.querySelector("[data-subject-scope-mode]"); const filtersWrap = root.querySelector("[data-subject-scope-filters]"); const popovers = Array.from(root.querySelectorAll("[data-subject-scope-popover]")); if (!modeSelect || !filtersWrap) return; function selectedOptions(select) { return Array.from(select.options || []).filter((option) => option.selected); } function closeMenus(exceptDetails) { popovers.forEach((popover) => { const details = popover.querySelector("details"); if (details && details !== exceptDetails) details.open = false; }); } function syncPopover(popover) { const select = popover.querySelector("select"); const summary = popover.querySelector("[data-filter-summary]"); const optionsWrap = popover.querySelector("[data-filter-options]"); if (!select || !summary || !optionsWrap) return; const selected = selectedOptions(select); popover.classList.toggle("is-active", selected.length > 0); if (!selected.length) { summary.textContent = "Qualsevol"; } else if (selected.length === 1) { summary.textContent = selected[0].textContent.trim(); } else { summary.textContent = `${selected.length} seleccionats`; } Array.from(optionsWrap.querySelectorAll("[data-option-value]")).forEach((label) => { const input = label.querySelector("input"); const isSelected = !!(input && input.checked); label.classList.toggle("is-selected", isSelected); }); } function clearSelect(select) { Array.from(select.options || []).forEach((option) => { option.selected = false; }); select.dispatchEvent(new Event("change", { bubbles: true })); } function renderPopover(popover) { const select = popover.querySelector("select"); const optionsWrap = popover.querySelector("[data-filter-options]"); const clearButton = popover.querySelector("[data-subject-scope-clear]"); const details = popover.querySelector("details"); if (!select || !optionsWrap) return; optionsWrap.innerHTML = ""; const options = Array.from(select.options || []); if (!options.length) { const empty = document.createElement("div"); empty.className = "groups-filter-empty"; empty.textContent = "Sense opcions"; optionsWrap.appendChild(empty); } else { options.forEach((option) => { const label = document.createElement("label"); label.className = "groups-filter-option"; label.setAttribute("data-option-value", option.value); const input = document.createElement("input"); input.type = "checkbox"; input.checked = option.selected; input.value = option.value; const text = document.createElement("span"); text.textContent = option.textContent.trim(); label.appendChild(input); label.appendChild(text); optionsWrap.appendChild(label); input.addEventListener("change", () => { option.selected = input.checked; select.dispatchEvent(new Event("change", { bubbles: true })); }); }); } select.addEventListener("change", () => { Array.from(optionsWrap.querySelectorAll("[data-option-value]")).forEach((label) => { const input = label.querySelector("input"); const option = Array.from(select.options || []).find((item) => item.value === label.getAttribute("data-option-value")); if (input && option) input.checked = option.selected; }); syncPopover(popover); }); if (clearButton) { clearButton.addEventListener("click", () => { clearSelect(select); }); } if (details) { details.addEventListener("toggle", () => { if (details.open) closeMenus(details); }); } syncPopover(popover); } function setFiltersEnabled(enabled, clearWhenDisabled) { filtersWrap.hidden = !enabled; popovers.forEach((popover) => { const select = popover.querySelector("select"); const details = popover.querySelector("details"); if (select) { select.disabled = !enabled; if (!enabled && clearWhenDisabled) clearSelect(select); } if (!enabled && details) details.open = false; syncPopover(popover); }); } popovers.forEach(renderPopover); modeSelect.addEventListener("change", () => { setFiltersEnabled(modeSelect.value === "filters", true); }); document.addEventListener("click", (event) => { if (!root.contains(event.target)) closeMenus(null); }); setFiltersEnabled(modeSelect.value === "filters", false); } initSubjectScopeFilters(assignmentForm);
