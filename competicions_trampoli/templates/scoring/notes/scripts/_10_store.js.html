function subjectStoreIdForSubject(subject, fallbackId){ const kind = String((subject && subject.subject_kind) || "inscripcio"); const sid = String((subject && subject.subject_id) || fallbackId || ""); return `${kind}:${sid}`; } function subjectStoreIdForId(insId){ const subject = SUBJECTS_BY_ID[String(insId)] || {}; return subjectStoreIdForSubject(subject, insId); } function normalizeScorePhaseId(faseId){ const value = faseId === undefined ? active.closest("table[data-scoring-table='1']") : null; if(table && table.dataset.faseId !== undefined) return table.dataset.faseId || ""; const payload = NOTES_STATE.renderingPayload || NOTES_STATE.activePayload || null; return payload?.context?.fase_id || ""; } function keyFor(insId, ex, appId, faseId){ const base = `${subjectStoreIdForId(insId)}|${ex}|${appId}`; const phase = normalizeScorePhaseId(faseId); return phase ? `${base}|${phase}` : base; } function keyForSubject(subjectKind, subjectId, ex, appId, faseId){ const base = `${String(subjectKind)}:${String(subjectId)}|${ex}|${appId}`; const phase = normalizeScorePhaseId(faseId); return phase ?
