"Tancar configuracio" : "Obrir configuracio"; } } function toggleComputedDetail(mainRow){ const btn = mainRow?.querySelector('[data-action="toggle-computed-detail"]'); const isOpen = btn?.getAttribute("aria-expanded") === "true"; setComputedDetailOpen(mainRow, !isOpen); scheduleUiStateSave(); } function shouldIgnoreComputedRowToggle(target){ if(!target) return true; const interactive = target.closest("button, select, textarea, a, label"); if(interactive) return true; const input = target.closest("input"); return !!(input && input.getAttribute("data-k") !== "formula_preview"); } function selectedText(mainRow, key){ const el = mainRow.querySelector(`[data-k="${key}"]`); if(!el) return ""; if(el.tagName === "SELECT"){ return el.options[el.selectedIndex]?.textContent?.trim() || el.value || ""; } return (el.value || "").trim(); } function compactParts(parts){ return parts.map(part => String(part || "").trim()).filter(Boolean).join(" · "); } function shortFormulaSummary(mainRow, formula){ const preset = mainRow.querySelector('[data-k="preset"]')?.value || "manual"; if(preset === "manual"){ return "Manual"; } if(preset === "exec_trampoli"){ return compactParts([selectedText(mainRow, "et_source"), "per jutge"]); } if(preset === "select_sum_guided"){ return compactParts([ selectedText(mainRow, "ss_source"), selectedText(mainRow, "ss_select"), selectedText(mainRow, "ss_agg"), ]); } if(preset === "member_treatment_guided"){ return compactParts([ selectedText(mainRow, "mt_source"), selectedText(mainRow, "mt_select"), selectedText(mainRow, "mt_agg"), ]); } if(preset === "row_compute" || preset === "row_agregation" || preset === "column_compute"){ const source = selectedText(mainRow, "rc_source"); const expr = selectedText(mainRow, "rc_item_expr"); const internalAgg = selectedText(mainRow, "rc_internal_agg"); const finalAgg = preset === "row_agregation" ?
